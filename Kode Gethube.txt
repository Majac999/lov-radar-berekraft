CACHE_DIR = "tekst_cache"



def hent_lovtekst(url):

try:

headers = {'User-Agent': 'Mozilla/5.0 (LovRadar Bot)'}

r = requests.get(url, headers=headers, timeout=10)

r.encoding = 'utf-8'

return r.text

except Exception as e:

print(f"Feil ved henting av {url}: {e}")

return ""
